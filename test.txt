Чтобы не копировать структуру class Program и Main для каждой задачи, я подготовил для тебя полный код. В нем задачи разделены комментариями. Ты можешь запускать их по очереди или просто вставить целиком в свой проект.

Главное, не забудь вводить значения для переменной a, когда программа этого просит.
C#

using System;

class Program
{
    static void Main()
    {
        // 1. Натуральное число: четность и окончание на 7
        Console.Write("Введите натуральное число: ");
        int n = int.Parse(Console.ReadLine());
        Console.WriteLine(n % 2 == 0 ? "Четное" : "Нечетное");
        Console.WriteLine(n % 10 == 7 ? "Оканчивается на 7" : "Не оканчивается на 7");

        // 2. Двузначное число: сравнение цифр
        Console.Write("\nВведите двузначное число: ");
        int n2 = int.Parse(Console.ReadLine());
        int d2_1 = n2 / 10;
        int d2_2 = n2 % 10;
        if (d2_1 > d2_2) Console.WriteLine("Первая цифра больше");
        else if (d2_2 > d2_1) Console.WriteLine("Вторая цифра больше");
        else Console.WriteLine("Цифры одинаковы");

        // 3. Квадрат числа и учетверенная сумма кубов (для 48 — true)
        double square48 = Math.Pow(n2, 2);
        double sumCubes48 = 4 * (Math.Pow(d2_1, 3) + Math.Pow(d2_2, 3));
        Console.WriteLine(square48 == sumCubes48 ? "Равен" : "Не равен");

        // 4. Сумма цифр двузначного числа
        Console.Write("Введите число a: ");
        int a = int.Parse(Console.ReadLine());
        int sum2 = d2_1 + d2_2;
        Console.WriteLine(sum2 >= 10 ? "Сумма двузначная" : "Сумма однозначная");
        Console.WriteLine(sum2 > a ? "Сумма больше a" : "Сумма не больше a");

        // 5. Кратность суммы цифр
        Console.WriteLine(sum2 % 3 == 0 ? "Сумма кратна 3" : "Сумма не кратна 3");
        Console.WriteLine(a != 0 && sum2 % a == 0 ? "Сумма кратна a" : "Сумма не кратна a");

        // 6. Трехзначное число: Палиндром
        Console.Write("\nВведите трехзначное число: ");
        int n3 = int.Parse(Console.ReadLine());
        int d3_1 = n3 / 100;
        int d3_2 = (n3 / 10) % 10;
        int d3_3 = n3 % 10;
        Console.WriteLine(d3_1 == d3_3 ? "Палиндром" : "Не палиндром");

        // 7. Какая из цифр трехзначного больше
        Console.WriteLine(d3_1 > d3_3 ? "1-я больше последней" : "3-я больше или равна 1-й");
        Console.WriteLine(d3_1 > d3_2 ? "1-я больше 2-й" : "2-я больше или равна 1-й");

        // 8. Квадрат и сумма кубов (трехзначное)
        Console.WriteLine(Math.Pow(n3, 2) == (Math.Pow(d3_1, 3) + Math.Pow(d3_2, 3) + Math.Pow(d3_3, 3)) ? "Равен" : "Не равен");

        // 9. Сложные проверки для трехзначного
        int sum3 = d3_1 + d3_2 + d3_3;
        int prod3 = d3_1 * d3_2 * d3_3;
        Console.WriteLine(sum3 >= 10 ? "Сумма двузначная" : "Сумма не двузначная");
        Console.WriteLine(prod3 >= 100 && prod3 <= 999 ? "Произведение трехзначное" : "Произведение не трехзначное");
        Console.WriteLine(prod3 > a ? "Произведение больше a" : "Произведение не больше a");
        Console.WriteLine(sum3 % 5 == 0 ? "Сумма кратна 5" : "Сумма не кратна 5");

        // 10. Одинаковые цифры в трехзначном
        Console.WriteLine((d3_1 == d3_2 && d3_2 == d3_3) ? "Все одинаковые" : "Не все одинаковые");
        Console.WriteLine((d3_1 == d3_2 || d3_2 == d3_3 || d3_1 == d3_3) ? "Есть одинаковые" : "Все разные");

        // 11. Четырехзначное число
        Console.Write("\nВведите четырехзначное число: ");
        int n4 = int.Parse(Console.ReadLine());
        int d4_1 = n4 / 1000, d4_2 = (n4 / 100) % 10, d4_3 = (n4 / 10) % 10, d4_4 = n4 % 10;
        Console.WriteLine((d4_1 + d4_2) == (d4_3 + d4_4) ? "Суммы половин равны" : "Суммы половин не равны");
        Console.WriteLine((d4_1 + d4_2 + d4_3 + d4_4) % 3 == 0 ? "Сумма кратна 3" : "Не кратна 3");
        Console.WriteLine((d4_1 * d4_2 * d4_3 * d4_4) % 4 == 0 ? "Произведение кратно 4" : "Не кратно 4");

        // 12. Конец на четную/нечетную
        Console.WriteLine((n % 10) % 2 == 0 ? "Кончается на четную" : "Кончается на нечетную");

        // 13. Делители a и b
        Console.Write("\nВведите число b: ");
        int b = int.Parse(Console.ReadLine());
        if (a != 0) Console.WriteLine(b % a == 0 ? "a является делителем b" : "a не делитель b");
        if (b != 0) Console.WriteLine(a % b == 0 ? "b является делителем a" : "b не делитель a");

        Console.ReadKey();
    }

}
